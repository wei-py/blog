### 防抖和节流

```js
// 防抖 用户输入 窗口调整
function debounce(callback, delay) {
  let timer = null;
  return function(...args) {
    if (timer) {
      clearTimeout(timer);
    }
    timer = setTimeout(() => {
      callback.apply(this, args);
    }, delay);
  }
}

// 节流 鼠标移动 滚动事件
function throttle(callback, delay) {
  let lastTime = Date.now();
  return function(...args) {
    const now = Date.now();
    if (now - lastTime >= delay) {
      callback.apply(this, args);
      lastTime = now;
    }
  }
}
```
